//第一次出现
function binarySearch(a, target) {
  var start = 0, end = a.length - 1;
 
  while(start <= end) {
    var mid = ~~((start + end) >> 1);
    if (a[mid] >= target)
      end = mid - 1;
    else 
      start = mid + 1;
  }
 
  return a[start] === target ? start : -1;
}
